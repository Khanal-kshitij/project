#include <iostream>
#include <string>
#include <fstream>
#ifdef _WIN32
#include <winsock2.h>
#pragma comment(lib, "Ws2_32.lib")
#else
#include <sys/socket.h>
#include <netinet/in.h>
#include <arpa/inet.h>
#include <unistd.h>
#endif
#include <cstring>

void send_command(int sock, const std::string& command) {
    std::string msg = command + "\n";
    send(sock, msg.c_str(), msg.size(), 0);
}

std::string receive_response(int sock) {
    char buffer[1024];
    std::string response;
    int bytes_received = recv(sock, buffer, sizeof(buffer) - 1, 0);
    if (bytes_received > 0) {
        buffer[bytes_received] = '\0';
        response = buffer;
    }
    return response;
}

bool upload_file(int sock, const std::string& filename) {
    std::ifstream infile(filename, std::ios::binary | std::ios::ate);
    if (!infile) {
        std::cout << "Error: Cannot open file " << filename << "\n";
        return false;
    }

    // Send file size
    uint32_t file_size = infile.tellg();
    uint32_t file_size_net = htonl(file_size);
    infile.seekg(0);
    if (send(sock, reinterpret_cast<char*>(&file_size_net), sizeof(file_size_net), 0) != sizeof(file_size_net)) {
        std::cout << "Error: Failed to send file size\n";
        return false;
    }

    // Send file data
    char buffer[1024];
    while (infile) {
        infile.read(buffer, sizeof(buffer));
        size_t bytes_read = infile.gcount();
        if (bytes_read > 0) {
            if (send(sock, buffer, bytes_read, 0) <= 0) {
                std::cout << "Error: File transfer interrupted\n";
                return false;
            }
        }
    }
    infile.close();
    std::cout << receive_response(sock) << "\n";
    return true;
}

bool download_file(int sock, const std::string& filename) {
    // Receive file size
    uint32_t file_size;
    if (recv(sock, reinterpret_cast<char*>(&file_size), sizeof(file_size), 0) != sizeof(file_size)) {
        std::cout << "Error: Failed to receive file size\n";
        return false;
    }
    file_size = ntohl(file_size);

    // Receive file data
    std::ofstream outfile(filename, std::ios::binary);
    if (!outfile) {
        std::cout << "Error: Cannot create file " << filename << "\n";
        return false;
    }

    char buffer[1024];
    size_t total_received = 0;
    while (total_received < file_size) {
        int bytes_received = recv(sock, buffer, std::min(sizeof(buffer), file_size - total_received), 0);
        if (bytes_received <= 0) {
            std::cout << "Error: File transfer interrupted\n";
            outfile.close();
            std::filesystem::remove(filename);
            return false;
        }
        outfile.write(buffer, bytes_received);
        total_received += bytes_received;
    }
    outfile.close();
    std::cout << receive_response(sock) << "\n";
    return true;
}

int main() {
#ifdef _WIN32
    WSADATA wsaData;
    if (WSAStartup(MAKEWORD(2, 2), &wsaData) != 0) {
        std::cerr << "WSAStartup failed\n";
        return 1;
    }
#endif

    // Create socket
    int sock = socket(AF_INET, SOCK_STREAM, 0);
    if (sock < 0) {
        std::cerr << "Socket creation failed\n";
        return 1;
    }

    // Connect to server
    sockaddr_in server_addr;
    server_addr.sin_family = AF_INET;
    server_addr.sin_port = htons(8888);
#ifdef _WIN32
    server_addr.sin_addr.s_addr = inet_addr("127.0.0.1");
#else
    inet_pton(AF_INET, "127.0.0.1", &server_addr.sin_addr);
#endif
    if (connect(sock, (sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
        std::cerr << "Connection failed\n";
        return 1;
    }

    // Authenticate
    send_command(sock, "AUTH user pass");
    std::cout << receive_response(sock) << "\n";

    // Command-line interface
    std::string command;
    while (true) {
        std::cout << "Enter command (UPLOAD <filename>, DOWNLOAD <filename>, LIST, EXIT): ";
        std::getline(std::cin, command);
        if (command == "EXIT") {
            break;
        }

        send_command(sock, command);
        if (command == "LIST") {
            std::cout << receive_response(sock);
        } else if (command.find("UPLOAD ") == 0) {
            std::string filename = command.substr(7);
            upload_file(sock, filename);
        } else if (command.find("DOWNLOAD ") == 0) {
            std::string filename = command.substr(9);
            download_file(sock, filename);
        } else {
            std::cout << receive_response(sock) << "\n";
        }
    }

#ifdef _WIN32
    closesocket(sock);
    WSACleanup();
#else
    close(sock);
#endif
    return 0;
}
